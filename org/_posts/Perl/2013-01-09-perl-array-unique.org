*** 例子:
#+begin_src perl -n :results output
  use Modern::Perl;
  use Data::Dumper;
  
  my @a = (
      { a => 'b'},
      { a => 'c'},
      { a => 'b'},
  );
  
  say "-----------------------";
  
  my $e = [ keys %{ {map {$_ => 1}  map {$_->{a} } @a } } ];
  
  say Dumper $e;
  
#+end_src
