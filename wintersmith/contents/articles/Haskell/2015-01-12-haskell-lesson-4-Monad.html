Monad

Monad is an Applicative Functor 

class Monad m where
  return = a -> m a
  (>>=) :: m a -> (a -> m b) -> m b
  (>>) :: ma -> m b -> m b
  (>>) = (\_ -> x)
  fail :: String -> m a
  fail string = error string

do

chain nested Monad toggether. 

ret = Just 10 >>= (\x -> Just 20 >>= (\y -> Just 30 >>= (\z -> Just (x + y + z)))))
ret = do
   x <- Just 10
   y <- Just 30
   z <- Just 20
   return (x + y + z)
